233  PLAYRHO_CONSTEXPR inline Length2D Simplex::CalcSearchDirection(const Edges& simplexEdges) noexcept
234  {
235  assert((simplexEdges.size() == 1) || (simplexEdges.size() == 2));
236  switch (simplexEdges.size())
237  {
238  case 1:
239  {
240  return -GetPointDelta(simplexEdges[0]);
241  }
242 
243  case 2:
244  {
245  const auto e12 = GetPointDelta(simplexEdges[1]) - GetPointDelta(simplexEdges[0]);
246  const auto e0 = GetPointDelta(simplexEdges[0]);
247  const auto sgn = Cross(e12, -e0);
248  // If sgn > 0, then origin is left of e12, else origin is right of e12.
249  return (sgn > Real{0} * SquareMeter)? GetRevPerpendicular(e12): GetFwdPerpendicular(e12);
250  }
251 
252  default:
253  return Length2D(0, 0);
254  }
255  }

257  inline Real Simplex::CalcMetric(const Edges& simplexEdges)
258  {
259  assert(simplexEdges.size() < 4);
260  switch (simplexEdges.size())
261  {
262  case 0: return Real{0};
263  case 1: return Real{0};
264  case 2:
265  {
266  const auto delta = GetPointDelta(simplexEdges[1]) - GetPointDelta(simplexEdges[0]);
267  return Sqrt(GetLengthSquared(GetVec2(delta)));
268  }
269  case 3:
270  {
271  const auto delta10 = GetPointDelta(simplexEdges[1]) - GetPointDelta(simplexEdges[0]);
272  const auto delta20 = GetPointDelta(simplexEdges[2]) - GetPointDelta(simplexEdges[0]);
273  return Cross(GetVec2(delta10), GetVec2(delta20));
274  }
275  default: break; // should not be reached
276  }
277  return Real{0};
278  }
