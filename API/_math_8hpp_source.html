199 constexpr inline bool almost_equal(float x, float y, int ulp = 2)
200 {
201  // From http://en.cppreference.com/w/cpp/types/numeric_limits/epsilon :
202  // "the machine epsilon has to be scaled to the magnitude of the values used
203  // and multiplied by the desired precision in ULPs (units in the last place)
204  // unless the result is subnormal".
205  // Where "subnormal" means almost zero.
206  //
207  return (Abs(x - y) < (std::numeric_limits<float>::epsilon() * Abs(x + y) * ulp)) || almost_zero(x - y);
208 }

338 template <class T1, class T2>
339 constexpr inline auto Cross(const T1 a, const T2 b) noexcept
340 {
341  // Both vectors of same direction...
342  // If a = Vec2{1, 2} and b = Vec2{1, 2} then: a x b = 1 * 2 - 2 * 1 = 0.
343  // If a = Vec2{1, 2} and b = Vec2{2, 4} then: a x b = 1 * 4 - 2 * 2 = 0.
344  //
345  // Vectors at +/- 90 degrees of each other...
346  // If a = Vec2{1, 2} and b = Vec2{-2, 1} then: a x b = 1 * 1 - 2 * (-2) = 1 + 4 = 5.
347  // If a = Vec2{1, 2} and b = Vec2{2, -1} then: a x b = 1 * (-1) - 2 * 2 = -1 - 4 = -5.
348  //
349  // Vectors between 0 and 180 degrees of each other excluding 90 degrees...
350  // If a = Vec2{1, 2} and b = Vec2{-1, 2} then: a x b = 1 * 2 - 2 * (-1) = 2 + 2 = 4.
351  return (a.GetX() * b.GetY()) - (a.GetY() * b.GetX());
352 }

381 constexpr Vec3 Solve33(const Mat33& mat, const Vec3 b) noexcept
382 {
383  const auto dp = Dot(mat.ex, Cross(mat.ey, mat.ez));
384  const auto det = (dp != 0)? 1 / dp: dp;
385  const auto x = det * Dot(b, Cross(mat.ey, mat.ez));
386  const auto y = det * Dot(mat.ex, Cross(b, mat.ez));
387  const auto z = det * Dot(mat.ex, Cross(mat.ey, b));
388  return Vec3{x, y, z};
389 }

395 constexpr T Solve22(const Mat33& mat, const T b) noexcept
396 {

398  const auto det = (cp != 0)? 1 / cp: cp;
399  const auto x = det * (mat.ey.y * b.GetX() - mat.ey.x * b.GetY());
400  const auto y = det * (mat.ex.x * b.GetY() - mat.ex.y * b.GetX());

406 constexpr inline Mat33 GetInverse22(const Mat33& value) noexcept
407 {
408  const auto a = value.ex.x, b = value.ey.x, c = value.ex.y, d = value.ey.y;
409  auto det = (a * d) - (b * c);
410  if (det != Real{0})
411  {
412  det = Real{1} / det;
413  }
414  return Mat33{Vec3{det * d, -det * c, Real{0}}, Vec3{-det * b, det * a, 0}, Vec3{0, 0, 0}};
415 }

419 constexpr inline Mat33 GetSymInverse33(const Mat33& value) noexcept
420 {
421  auto det = Dot(value.ex, Cross(value.ey, value.ez));
422  if (det != Real{0})
423  {
424  det = Real{1} / det;
425  }
426 
427  const auto a11 = value.ex.x, a12 = value.ey.x, a13 = value.ez.x;
428  const auto a22 = value.ey.y, a23 = value.ez.y;
429  const auto a33 = value.ez.z;
430 
431  const auto ex_y = det * (a13 * a23 - a12 * a33);
432  const auto ey_z = det * (a13 * a12 - a11 * a23);
433  const auto ex_z = det * (a12 * a23 - a13 * a22);
434 
435  return Mat33{
436  Vec3{det * (a22 * a33 - a23 * a23), ex_y, ex_z},
437  Vec3{ex_y, det * (a11 * a33 - a13 * a13), ey_z},
438  Vec3{ex_z, ey_z, det * (a11 * a22 - a12 * a12)}
439  };
