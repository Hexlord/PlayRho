501 inline Body::FlagsType Body::GetFlags(const BodyType type) noexcept
502 {
503  auto flags = FlagsType{0};
504  switch (type)
505  {
506  case BodyType::Dynamic: flags |= (e_velocityFlag|e_accelerationFlag); break;
507  case BodyType::Kinematic: flags |= (e_impenetrableFlag|e_velocityFlag); break;
508  case BodyType::Static: flags |= (e_impenetrableFlag); break;
509  }
510  return flags;
511 }

513 inline BodyType Body::GetType() const noexcept
514 {
515  switch (m_flags & (e_accelerationFlag|e_velocityFlag))
516  {
517  case e_velocityFlag|e_accelerationFlag: return BodyType::Dynamic;
518  case e_velocityFlag: return BodyType::Kinematic;
519  default: break; // handle case 0 this way so compiler doesn't warn of no default handling.
520  }
521  return BodyType::Static;
522 }

858 inline void SetForce(Body& body, const Force2D force, const Length2D point) noexcept
859 {
860  const auto linAccel = LinearAcceleration2D{force * body.GetInvMass()};
861  const auto invRotI = body.GetInvRotInertia();
862  const auto dp = point - body.GetWorldCenter();
863  const auto cp = Torque{Cross(dp, force) / Radian};
864  const auto angAccel = AngularAcceleration{cp * invRotI};
865  body.SetAcceleration(linAccel, angAccel);
866 }

875 inline void ApplyForce(Body& body, const Force2D force, const Length2D point) noexcept
876 {
877  // Torque is L^2 M T^-2 QP^-1.
878  const auto linAccel = LinearAcceleration2D{force * body.GetInvMass()};
879  const auto invRotI = body.GetInvRotInertia(); // L^-2 M^-1 QP^2
880  const auto dp = Length2D{point - body.GetWorldCenter()}; // L
881  const auto cp = Torque{Cross(dp, force) / Radian}; // L * M L T^-2 is L^2 M T^-2
882  // L^2 M T^-2 QP^-1 * L^-2 M^-1 QP^2 = QP T^-2;
883  const auto angAccel = AngularAcceleration{cp * invRotI};
884  body.SetAcceleration(body.GetLinearAcceleration() + linAccel,
885  body.GetAngularAcceleration() + angAccel);
886 }
