93  template <typename T>
94  static constexpr value_type GetFromFloat(T val) noexcept
95  {
96  static_assert(std::is_floating_point<T>::value, "floating point value required");
97  // Note: std::isnan(val) *NOT* constant expression, so can't use here!
98  return !(val <= 0 || val >= 0)? GetNaN().m_value:
99  (val > static_cast<long double>(GetMax()))? GetInfinity().m_value:
100  (val < static_cast<long double>(GetLowest()))? GetNegativeInfinity().m_value:
101  static_cast<value_type>(val * ScaleFactor);
102  }

104  template <typename T>
105  static constexpr value_type GetFromSignedInt(T val) noexcept
106  {
107  static_assert(std::is_integral<T>::value, "integral value required");
108  static_assert(std::is_signed<T>::value, "must be signed");
109  return (val > (GetMax().m_value / ScaleFactor))? GetInfinity().m_value:
110  (val < (GetLowest().m_value / ScaleFactor))? GetNegativeInfinity().m_value:
111  static_cast<value_type>(val * ScaleFactor);
112  }
113 
114  template <typename T>
115  static constexpr value_type GetFromUnsignedInt(T val) noexcept
116  {
117  static_assert(std::is_integral<T>::value, "integral value required");
118  static_assert(!std::is_signed<T>::value, "must be unsigned");
119  const auto max = static_cast<unsigned_wider_type>(GetMax().m_value / ScaleFactor);
120  return (val > max)? GetInfinity().m_value: static_cast<value_type>(val) * ScaleFactor;
121  }

208  constexpr CmpResult Compare(const Fixed other) const noexcept
209  {
210  if (isnan() || other.isnan())
211  {

213  }
214  if (m_value < other.m_value)
215  {
216  return CmpResult::LessThan;
217  }

219  {

221  }

300  constexpr Fixed& operator+= (Fixed val) noexcept
301  {
302  if (isnan() || val.isnan()
303  || ((m_value == GetInfinity().m_value) && (val.m_value == GetNegativeInfinity().m_value))
304  || ((m_value == GetNegativeInfinity().m_value) && (val.m_value == GetInfinity().m_value))
305  )
306  {
307  *this = GetNaN();
308  }
309  else if (val.m_value == GetInfinity().m_value)
310  {
311  m_value = GetInfinity().m_value;
312  }
313  else if (val.m_value == GetNegativeInfinity().m_value)
314  {
315  m_value = GetNegativeInfinity().m_value;
316  }
317  else if (isfinite() && val.isfinite())
318  {
319  const auto result = wider_type{m_value} + val.m_value;
320  if (result > GetMax().m_value)
321  {
322  // overflow from max
323  m_value = GetInfinity().m_value;
324  }
325  else if (result < GetLowest().m_value)
326  {
327  // overflow from lowest
328  m_value = GetNegativeInfinity().m_value;
329  }
330  else
331  {
332  m_value = static_cast<value_type>(result);
333  }
334  }
335  return *this;
336  }

338  constexpr Fixed& operator-= (Fixed val) noexcept
339  {
340  if (isnan() || val.isnan()
341  || ((m_value == GetInfinity().m_value) && (val.m_value == GetInfinity().m_value))
342  || ((m_value == GetNegativeInfinity().m_value) && (val.m_value == GetNegativeInfinity().m_value))

344  {
345  *this = GetNaN();

376  constexpr Fixed& operator*= (Fixed val) noexcept
377  {
378  if (isnan() || val.isnan())
379  {
380  *this = GetNaN();
381  }
382  else if (!isfinite() || !val.isfinite())
383  {
384  if (m_value == 0 || val.m_value == 0)
385  {
386  *this = GetNaN();
387  }
388  else
389  {
390  *this = ((m_value > 0) != (val.m_value > 0))? -GetInfinity(): GetInfinity();
391  }
392  }
393  else
394  {
395  const auto product = wider_type{m_value} * wider_type{val.m_value};
396  const auto result = product / ScaleFactor;
397 
398  if (product != 0 && result == 0)
399  {
400  // underflow
401  m_value = static_cast<value_type>(result);
402  }
403  else if (result > GetMax().m_value)
404  {
405  // overflow from max
406  m_value = GetInfinity().m_value;
407  }
408  else if (result < GetLowest().m_value)
409  {
410  // overflow from lowest
411  m_value = GetNegativeInfinity().m_value;
412  }
413  else
414  {
415  m_value = static_cast<value_type>(result);
416  }
417  }
418  return *this;
419  }

421  constexpr Fixed& operator/= (Fixed val) noexcept
422  {
423  if (isnan() || val.isnan())
424  {
425  *this = GetNaN();
426  }
427  else if (!isfinite() && !val.isfinite())
428  {
429  *this = GetNaN();
430  }
431  else if (!isfinite())
432  {
433  *this = ((m_value > 0) != (val.m_value > 0))? -GetInfinity(): GetInfinity();
434  }
435  else if (!val.isfinite())
436  {
437  *this = 0;
438  }
439  else
440  {
441  const auto product = wider_type{m_value} * ScaleFactor;
442  const auto result = product / val.m_value;
443 
444  if (product != 0 && result == 0)
445  {
446  // underflow
447  m_value = static_cast<value_type>(result);
448  }
449  else if (result > GetMax().m_value)
450  {
451  // overflow from max
452  m_value = GetInfinity().m_value;
453  }
454  else if (result < GetLowest().m_value)
455  {

458  }
459  else
