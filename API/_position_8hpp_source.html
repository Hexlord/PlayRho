105  inline Position GetPosition(const Position pos0, const Position pos1, const Real beta) noexcept
106  {
107  // Note: have to be careful how this is done.
108  // If pos0 == pos1 then return value should always be equal to pos0 too.
109  // But if Real is float, pos0 * (1 - beta) + pos1 * beta can fail this requirement.
110  // Meanwhile, pos0 + (pos1 - pos0) * beta always works.
111 
112  // pos0 * (1 - beta) + pos1 * beta
113  // pos0 - pos0 * beta + pos1 * beta
114  // pos0 + (pos1 * beta - pos0 * beta)
115  // pos0 + (pos1 - pos0) * beta
116  return pos0 + (pos1 - pos0) * beta;
117  }
