82  template <typename T, LoValueCheck lo, HiValueCheck hi>
83  class BoundedValue
84  {
85  public:
86  using value_type = T;
87  using remove_pointer_type = typename std::remove_pointer<T>::type;
88  using exception_type = InvalidArgument;
89  using this_type = BoundedValue<value_type, lo, hi>;
90 
91  static constexpr LoValueCheck GetLoCheck() { return lo; }
92  static constexpr HiValueCheck GetHiCheck() { return hi; }
93 
94  static constexpr void DoLoCheck(value_type value)
95  {
96  switch (GetLoCheck())
97  {
98  case LoValueCheck::Any:
99  return;
100  case LoValueCheck::AboveZero:
101  if (!(value > value_type(0)))
102  {
103  throw exception_type{"value not > 0"};
104  }
105  return;
106  case LoValueCheck::ZeroOrMore:
107  if (!(value >= value_type(0)))
108  {
109  throw exception_type{"value not >= 0"};
110  }
111  return;
112  case LoValueCheck::AboveNegInf:
113  CheckIfAboveNegInf<T>(value);
114  return;
115  case LoValueCheck::NonZero:
116  if (value == static_cast<value_type>(0))
117  {
118  throw exception_type{"value may not be 0"};
119  }
120  return;
121  }
122  }
123 
124  static constexpr void DoHiCheck(value_type value)
125  {
126  switch (GetHiCheck())
127  {
128  case HiValueCheck::Any:
129  return;
130  case HiValueCheck::BelowZero:
131  if (!(value < value_type(0)))
132  {
133  throw exception_type{"value not < 0"};
134  }
135  return;
136  case HiValueCheck::ZeroOrLess:
137  if (!(value <= value_type(0)))
138  {
139  throw exception_type{"value not <= 0"};
140  }
141  return;
142  case HiValueCheck::OneOrLess:
143  if (!ValueCheckHelper<value_type>::has_one)
144  {
145  throw exception_type{"value's type does not have a trivial 1"};
146  }
147  if (!(value <= ValueCheckHelper<value_type>::one()))
148  {
149  throw exception_type{"value not <= 1"};
150  }
151  return;
152  case HiValueCheck::BelowPosInf:
153  if (std::numeric_limits<value_type>::has_infinity)
154  {
155  if (!(value < +std::numeric_limits<value_type>::infinity()))
156  {
157  throw exception_type{"value not < +inf"};;
158  }
159  }
160  return;
161  }
162  }
