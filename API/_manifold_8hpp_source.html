473  inline void Manifold::AddPoint(cf_t type, sidx_t index, Length2D point) noexcept
474  {
475  assert(m_pointCount < MaxManifoldPoints);
476  switch (m_type)
477  {
478  case e_unset:
479  break;
480  case e_circles:
481  break;
482  case e_faceA:
483  m_points[m_pointCount].localPoint = point;
484  m_points[m_pointCount].contactFeature.typeB = type;
485  m_points[m_pointCount].contactFeature.indexB = index;
486  ++m_pointCount;
487  break;
488  case e_faceB:
489  m_points[m_pointCount].localPoint = point;
490  m_points[m_pointCount].contactFeature.typeA = type;
491  m_points[m_pointCount].contactFeature.indexA = index;
492  ++m_pointCount;
493  break;
494  }
