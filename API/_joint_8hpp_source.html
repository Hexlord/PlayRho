270 inline BodyConstraintPtr& At(std::vector<BodyConstraintPair>& container, const Body* key)
271 {
272  auto last = std::end(container);
273  auto first = std::begin(container);
274  first = std::lower_bound(first, last, key, [](const BodyConstraintPair &a, const Body* b){
275  return a.first < b;
276  });
277  if (first == last || key != (*first).first)
278  {
279  throw std::out_of_range{"invalid key"};
280  }
281  return (*first).second;
282 }
